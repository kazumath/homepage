TikZでアナログ時計

TikZ-3dplotで立方体を書きました. 見えない面は自動で破線で書かれるところがポイントです(*'ω'*)

コード

以下がコードです. ご自由にお使いください. 

\newcommand{\drawcube}[3][1]{% #1=length, \tdplotsetmaincoords{#2}{#3}\tdplotsetmaincoords{#2}{#3}\colorlet{maincolor}{cyan}\begin{tikzpicture}[scale=1.5,tdplot_main_coords,axis/.style={-stealth},mainline/.style={maincolor,very thick, opacity=0.7},invisibleline/.style={cyan,densely dashed},]\pgfmathsetmacro{\length}{#1}  % axis  \draw[axis](0,0,0)--(0.4*\length,0,0)node[pos=1.1]{\scriptsize$x$};  \draw[axis](0,0,0)--(0,0.4*\length,0)node[pos=1.1]{\scriptsize$y$};  \draw[axis](0,0,0)--(0,0,0.4*\length)node[pos=1.1]{\scriptsize$z$};  % vertices  \coordinate (A) at (0, 0, 0);  \coordinate (B) at (\length, 0, 0);  \coordinate (C) at (\length, \length, 0);  \coordinate (D) at (0, \length, 0);  \coordinate (E) at (0, 0, \length);  \coordinate (F) at (\length, 0, \length);  \coordinate (G) at (\length, \length, \length);  \coordinate (H) at (0, \length, \length);  % label  \foreach \P in{A,B,C,D,E,F,G,H}  \node[maincolor!50!black] at(\P){\P};  % camera vector: ({sin(\tdplotmaintheta)*sin(\tdplotmainphi)},{sin(\tdplotmaintheta)*cos(\tdplotmainphi)},{cos(\tdplotmaintheta)})  % inner product  \pgfmathparse{-cos(\tdplotmaintheta) >= 0}  \pgfmathsetmacro{\ABCDprod}{\pgfmathresult}  \pgfmathparse{sin(\tdplotmaintheta)*sin(\tdplotmainphi) >= 0}  \pgfmathsetmacro{\BCFGprod}{\pgfmathresult}  \pgfmathparse{-sin(\tdplotmaintheta)*cos(\tdplotmainphi) >= 0}  \pgfmathsetmacro{\CDHGprod}{\pgfmathresult}  \pgfmathparse{-sin(\tdplotmaintheta)*sin(\tdplotmainphi) >= 0}  \pgfmathsetmacro{\DAEHprod}{\pgfmathresult}  \pgfmathparse{cos(\tdplotmaintheta) >= 0}  \pgfmathsetmacro{\EFGHprod}{\pgfmathresult}  \pgfmathparse{sin(\tdplotmaintheta)*cos(\tdplotmainphi) >= 0}  \pgfmathsetmacro{\ABEFprod}{\pgfmathresult}  \tikzset{    ABCD/.style={insert path={(A)--(B)--(C)--(D)--cycle}},    ABEF/.style={insert path={(A)--(B)--(F)--(E)--cycle}},    BCFG/.style={insert path={(B)--(C)--(G)--(F)--cycle}},    CDHG/.style={insert path={(C)--(D)--(H)--(G)--cycle}},    EFGH/.style={insert path={(E)--(F)--(G)--(H)--cycle}},     DAEH/.style={insert path={(D)--(A)--(E)--(H)--cycle}},  }  \ifthenelse{\ABCDprod=1 \and \ABEFprod=1 \and \BCFGprod=1}{  % Case 1: ABCD, ABEF, BCFG visible.  \foreach \face in{ABCD, ABEF, BCFG}  \draw[mainline,\face];   \foreach \P in{E,G,D}  \draw[invisibleline](H)--(\P);  }{  \ifthenelse{\ABCDprod=1 \and \ABEFprod=1 \and \DAEHprod=1}{  % Case 2: ABCD, ABEF, DAEH visible.  \foreach \face in{ABCD, ABEF, DAEH}  \draw[mainline,\face];   \foreach \P in{F,H,C}  \draw[invisibleline](G)--(\P);  }{  \ifthenelse{\ABCDprod=1 \and \CDHGprod=1 \and \BCFGprod=1}{  % Case 3: ABCD, CDHG, BCFG visible.  \foreach \face in{ABCD, CDHG, BCFG}  \draw[mainline,\face];   \foreach \P in{A,F,H}  \draw[invisibleline](E)--(\P);  }{  \ifthenelse{\ABCDprod=1 \and \CDHGprod=1 \and \DAEHprod=1}{  % Case 4: ABCD, CDHG, DAEH visible.  \foreach \face in{ABCD, CDHG, DAEH}  \draw[mainline,\face];   \foreach \P in{E,G,B}  \draw[invisibleline](F)--(\P);  }{  \ifthenelse{\EFGHprod=1 \and \ABEFprod=1 \and \BCFGprod=1}{  % Case 5: EFGH, ABEF, BCFG visible.  \foreach \face in{EFGH, ABEF, BCFG}  \draw[mainline,\face];   \foreach \P in{A,H,C}  \draw[invisibleline](D)--(\P);  }{  \ifthenelse{\EFGHprod=1 \and \ABEFprod=1 \and \DAEHprod=1}{  % Case 6: EFGH, ABEF, DAEH visible.  \foreach \face in{EFGH, ABEF, DAEH}  \draw[mainline,\face];   \foreach \P in{G,B,D}  \draw[invisibleline](C)--(\P);  }{  \ifthenelse{\EFGHprod=1 \and \CDHGprod=1 \and \BCFGprod=1}{  % Case 7: EFGH, CDHG, BCFG visible.  \foreach \face in{EFGH, CDHG, BCFG}  \draw[mainline,\face];   \foreach \P in{E,B,D}  \draw[invisibleline](A)--(\P);  }{ \draw[invisibleline](B)--(\P);  }{  }}}}}}}}\end{tikzpicture}}
